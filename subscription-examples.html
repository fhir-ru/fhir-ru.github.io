Example of subscription (id = "example")

Raw XML

<Subscription xmlns="http://hl7.org/fhir">
  <text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml">[Put rendering here]</div>
  </text>

  <criteria value="/Observation?name=http://loinc.org|1975-2"/>
  <reason value="Monitor new neonatal function"/>
  <status value="requested"/>
  <channel>
    <type value="rest-hook"/>
    <url value="https://biliwatch.com/customers/mount-auburn-miu/on-result"/>
    <payload value="application/json"/>
    <header value="Authorization: Bearer secret-token-abc-123"/>
  </channel>  
  
</Subscription>
